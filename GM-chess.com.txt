(function() {
    // Function to replace the first <a> element
    function replaceFirstAnchorElement() {
        // Select the <a> element with a specific data-tooltip-target and href
        var oldElement = document.querySelector('a[data-tooltip-target="6"][href*="badge-diamond"]');
        if (oldElement) {
            // Create a new <a> element
            var newElement = document.createElement('a');
            newElement.className = 'user-chess-title-component'; // Set the class of the new element
            newElement.href = '/members/titled-players'; // Set the href of the new element
            newElement.target = '_blank'; // Make the link open in a new tab
            newElement.dataset.tooltipTarget = '12'; // Set a data attribute on the new element
            newElement.textContent = 'GM'; // Set the text content of the new element
            oldElement.parentNode.replaceChild(newElement, oldElement); // Replace the old element with the new one
            console.log('First <a> element successfully replaced.'); // Log success message
        } else {
            console.log('First <a> element not found.'); // Log message if the old element was not found
        }
    }
